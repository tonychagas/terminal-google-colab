# 4. 🚀 INICIAR O OLLAMA
import os
import subprocess
import threading
import time
import requests
import json

print("="*60)
print("🚀 INICIANDO SERVIDOR OLLAMA")
print("="*60)

os.environ["OLLAMA_HOST"] = "127.0.0.1:11434"
os.environ["OLLAMA_KEEP_ALIVE"] = "-1"
os.environ["OLLAMA_NUM_PARALLEL"] = "4"
os.environ["OLLAMA_CONTEXT_LENGTH"] = "32768"
os.environ["OLLAMA_FLASH_ATTENTION"] = "1"
os.environ["OLLAMA_DEBUG"] = "0"

def start_ollama():
    env = os.environ.copy()
    env["OLLAMA_HOST"] = "127.0.0.1:11434"
    subprocess.Popen(
        ["ollama", "serve"],
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL,
        env=env
    )

thread = threading.Thread(target=start_ollama, daemon=True)
thread.start()
time.sleep(15)

# 6. 🌐 CONFIGURAR NGROK
from pyngrok import ngrok

NGROK_TOKEN = "coloque_seu_token_arqui_do_ngrok"

print("="*60)
print("🌐 CONFIGURANDO NGROK")
print("="*60)

ngrok.set_auth_token(NGROK_TOKEN)
ngrok.kill()

# Cria o túnel com host_header para evitar erros 403
tunnel = ngrok.connect(addr="11434", proto="http", host_header="localhost:11434")
public_url = tunnel.public_url

print(f"✅ Ngrok configurado!")
print(f"🔗 URL PÚBLICA: {public_url}")

# 7. 🧪 TESTAR CONEXÃO
print("="*60)
print("🔍 TESTANDO CONEXÃO")
print("="*60)

# Teste local
print("\n1️⃣ Testando localhost:11434...")
try:
    r = requests.get("http://127.0.0.1:11434/api/tags", timeout=5)
    if r.status_code == 200:
        models = r.json().get('models', [])
        model_name = models[0]['name'] if models else 'N/A'
        print(f"✅ Localhost OK! Modelo: {model_name}")
    else:
        print(f"❌ Localhost: {r.status_code}")
except Exception as e:
    print(f"❌ Localhost: {e}")

# Teste via ngrok
print(f"\n2️⃣ Testando {public_url}/api/tags...")
headers = {"ngrok-skip-browser-warning": "true"}
try:
    r = requests.get(f"{public_url}/api/tags", headers=headers, timeout=10)
    if r.status_code == 200:
        models = r.json().get('models', [])
        model_name = models[0]['name'] if models else 'N/A'
        print(f"✅ Ngrok OK! Modelo: {model_name}")
    else:
        print(f"❌ Ngrok: {r.status_code}")
        print(f"Resposta: {r.text[:200]}")
except Exception as e:
    print(f"❌ Ngrok: {e}")
